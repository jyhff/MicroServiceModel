## 2. 部署架构

### 2.1 环境划分
| 环境 | 用途 | 配置 | 访问 |
|------|------|------|------|
| dev | 开发 | 2C4G * 3 | 内网 |
| test | 测试 | 4C8G * 5 | 内网 |
| staging | 预发布 | 8C16G * 5 | VPN |
| prod | 生产 | 16C32G * 10+ | 公网 |

### 2.2 部署流程
```
代码提交 → CI构建 → 镜像推送 → 部署到dev → 自动化测试 → 
部署到test → 集成测试 → 部署到staging → 验收测试 → 
审批 → 灰度发布 → 全量发布 → 监控
```

## 3. Docker部署

### 3.1 Dockerfile
```dockerfile
# Dockerfile
FROM mcr.microsoft.com/dotnet/aspnet:10.0 AS base
WORKDIR /app
EXPOSE 80
EXPOSE 443

FROM mcr.microsoft.com/dotnet/sdk:10.0 AS build
WORKDIR /src
COPY . .
RUN dotnet restore
RUN dotnet build -c Release -o /app/build

FROM build AS publish
RUN dotnet publish -c Release -o /app/publish

FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "LCH.MicroService.VideoService.dll"]
```

            path: /health

## 5. CI/CD流水线

### 5.1 GitHub Actions
```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write
    steps:
      - name: Checkout
        uses: actions/checkout@v3

      - name: Setup .NET
        uses: actions/setup-dotnet@v3
        with:
          dotnet-version: '10.0.x'

      - name: Restore
        run: dotnet restore

      - name: Build
        run: dotnet build --no-restore --configuration Release

      - name: Test
        run: dotnet test --no-build --verbosity normal

      - name: Setup Docker Buildx
        uses: docker/setup-buildx-action@v2

      - name: Login to Registry
        uses: docker/login-action@v2
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v4
        with:
          images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}

      - name: Build and push
        uses: docker/build-push-action@v4
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

  deploy-staging:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - name: Checkout
        uses: actions/checkout@v3

      - name: Setup kubectl
        uses: azure/setup-kubectl@v3

      - name: Setup Helm
        uses: azure/setup-helm@v3

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v2
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: us-west-2

      - name: Update kubeconfig
        run: aws eks update-kubeconfig --name bilibili-staging

      - name: Deploy to staging
        run: |
          helm upgrade --install bilibili ./helm-chart \
            --namespace bilibili-staging \
            --set image.tag=${{ github.sha }} \
            --set environment=staging \
            --wait --timeout 10m

      - name: Verify deployment
        run: kubectl rollout status deployment/video-service -n bilibili-staging

  deploy-production:
    needs: deploy-staging
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    environment: production
    steps:
      - name: Manual approval
        uses: trstringer/manual-approval@v1
        with:
          secret: ${{ github.TOKEN }}
          approvers: team-leads

      - name: Deploy to production
        run: |
          helm upgrade --install bilibili ./helm-chart \
            --namespace bilibili-prod \
            --set image.tag=${{ github.sha }} \
            --set environment=production \
            --wait --timeout 10m

      - name: Verify deployment
        run: kubectl rollout status deployment/video-service -n bilibili-prod

      - name: Smoke test
        run: |
          kubectl run smoke-test --rm -i --restart=Never \
            --image=curlimages/curl:latest \
            -- curl -f http://gateway/health
```

```yaml
# rules/alerts.yml
groups:
- name: bilibili-alerts
  rules:
  - alert: HighErrorRate
    expr: rate(http_requests_total{status=~"5.."}[5m]) > 0.01
    for: 5m
    labels:
      severity: critical
    annotations:
      summary: "High error rate detected"
      description: "Error rate is above 1%"

  - alert: ServiceDown
    expr: up == 0
    for: 1m
    labels:
      severity: critical
    annotations:
      summary: "Service is down"

  - alert: HighLatency
    expr: histogram_quantile(0.95, rate(http_request_duration_seconds_bucket[5m])) > 0.5
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "High latency detected"

  - alert: DatabaseConnections
    expr: pg_stat_activity_count > 100
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "Database connection pool nearing limit"
```

## 7. 备份恢复

### 7.1 数据库备份
```bash
#!/bin/bash
# backup.sh

DB_NAME="PlatformDB"
BACKUP_DIR="/backup/postgres"
DATE=$(date +%Y%m%d_%H%M%S)

# 全量备份
pg_dump -h postgres -U postgres -Fc $DB_NAME > $BACKUP_DIR/${DB_NAME}_${DATE}.dump

# 保留最近7天备份
find $BACKUP_DIR -name "*.dump" -mtime +7 -delete

# 上传到S3
aws s3 cp $BACKUP_DIR/${DB_NAME}_${DATE}.dump s3://bilibili-backup/postgres/
```

### 7.2 Redis备份
```bash
#!/bin/bash
# redis-backup.sh

redis-cli -h redis BGSAVE

# 等待RDB文件生成
sleep 10

# 复制到备份目录
cp /var/lib/redis/dump.rdb /backup/redis/redis_${DATE}.rdb

# 上传到S3
aws s3 cp /backup/redis/redis_${DATE}.rdb s3://bilibili-backup/redis/
```

### 7.3 灾难恢复
```bash
#!/bin/bash
# restore.sh

# 从S3下载备份
aws s3 cp s3://bilibili-backup/postgres/${BACKUP_FILE} /tmp/

# 恢复数据库
pg_restore -h postgres -U postgres -d PlatformDB -c /tmp/${BACKUP_FILE}

# 重启服务
kubectl rollout restart deployment/video-service -n bilibili-prod
```

## 8. 运维手册

### 8.1 常用命令
```bash
# 查看Pod状态
kubectl get pods -n bilibili-prod

# 查看日志
kubectl logs -f deployment/video-service -n bilibili-prod

# 进入容器
kubectl exec -it pod/video-service-xxx -n bilibili-prod -- /bin/bash

# 水平扩展
kubectl scale deployment video-service --replicas=5 -n bilibili-prod

# 滚动重启
kubectl rollout restart deployment/video-service -n bilibili-prod

# 查看资源使用
kubectl top pods -n bilibili-prod
```

### 8.2 故障排查
| 故障 | 症状 | 排查 | 解决 |
|------|------|------|------|
| 服务不可用 | 502/503错误 | 检查Pod状态、资源使用 | 重启服务、扩容 |
| 数据库连接超时 | API响应慢 | 检查连接池、慢查询 | 优化SQL、增加连接池 |
| Redis内存不足 | 缓存命中率下降 | 检查内存使用、键数量 | 清理过期键、扩容 |
| 消息队列积压 | 任务处理延迟 | 检查队列长度、消费者数 | 增加消费者、优化处理逻辑 |
