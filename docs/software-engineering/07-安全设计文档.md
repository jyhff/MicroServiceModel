# 安全设计文档 (Security Design)

## 1. 安全策略总览

### 1.1 安全目标
- **机密性**: 保护用户数据和系统配置
- **完整性**: 确保数据不被篡改
- **可用性**: 防止服务中断
- **可追溯性**: 记录所有安全事件

### 1.2 安全架构
```
┌─────────────────────────────────────────────────────────────┐
│                      安全层次                               │
├─────────────┬─────────────────────────────────────────────────┤
│  网络层     │  WAF + DDoS防护 + IP白名单                       │
├─────────────┼─────────────────────────────────────────────────┤
│  应用层     │  OAuth2.0 + JWT + 签名验证 + 防重放               │
├─────────────┼─────────────────────────────────────────────────┤
│  数据层     │  加密存储 + 脱敏展示 + 访问控制                   │
├─────────────┼─────────────────────────────────────────────────┤
│  运维层     │  最小权限 + 操作审计 + 安全扫描                   │
└─────────────┴─────────────────────────────────────────────────┘
```

### 2.1 JWT Token设计
```csharp
public class JwtTokenService
{
    public string GenerateAccessToken(User user)
    {
        var claims = new[]
        {
            new Claim(JwtRegisteredClaimNames.Sub, user.Id.ToString()),
            new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString()),
            new Claim("username", user.UserName),
            new Claim("level", user.Level.ToString()),
            new Claim("roles", string.Join(",", user.Roles)),
        };

        var key = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_configuration["Jwt:SecretKey"]));
        var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

        var token = new JwtSecurityToken(
            issuer: _configuration["Jwt:Issuer"],
            audience: _configuration["Jwt:Audience"],
            claims: claims,
            expires: DateTime.Now.AddHours(2),  // Access Token 2小时
            signingCredentials: creds
        );

        return new JwtSecurityTokenHandler().WriteToken(token);
    }

    public string GenerateRefreshToken()
    {
        var randomNumber = new byte[32];
        using (var rng = RandomNumberGenerator.Create())
        {
            rng.GetBytes(randomNumber);
            return Convert.ToBase64String(randomNumber);
        }
    }
}
```

```csharp
public class DeviceManager

    public async Task<bool> AddDevice(Guid userId, DeviceInfo device)

        }
        
        await _deviceRepository.AddAsync(device);
        return true;
    }
}
```

## 3. API安全

### 3.1 请求签名
```csharp
public class RequestSignatureMiddleware
{
    public async Task InvokeAsync(HttpContext context)
    {
        var timestamp = context.Request.Headers["X-Timestamp"].FirstOrDefault();
        var signature = context.Request.Headers["X-Signature"].FirstOrDefault();
        var nonce = context.Request.Headers["X-Nonce"].FirstOrDefault();
        
        if (string.IsNullOrEmpty(timestamp) || string.IsNullOrEmpty(signature))
        {
            context.Response.StatusCode = 401;
            await context.Response.WriteAsync("Missing signature");
            return;
        }
        
        // 检查时间戳（5分钟有效期）
        var requestTime = DateTimeOffset.FromUnixTimeMilliseconds(long.Parse(timestamp));
        if (DateTimeOffset.UtcNow - requestTime > TimeSpan.FromMinutes(5))
        {
            context.Response.StatusCode = 401;
            await context.Response.WriteAsync("Request expired");
            return;
        }
        
        // 检查nonce（防重放）
        if (await _cache.ExistsAsync($"nonce:{nonce}"))
        {
            context.Response.StatusCode = 401;
            await context.Response.WriteAsync("Duplicate request");
            return;
        }
        await _cache.SetAsync($"nonce:{nonce}", "1", TimeSpan.FromMinutes(5));
        
        // 验证签名
        var expectedSignature = GenerateSignature(context.Request, timestamp, nonce);
        if (!CryptographicOperations.FixedTimeEquals(
            Convert.FromBase64String(signature),
            Convert.FromBase64String(expectedSignature)))
        {
            context.Response.StatusCode = 401;
            await context.Response.WriteAsync("Invalid signature");
            return;
        }
        
        await _next(context);
    }
    
    private string GenerateSignature(HttpRequest request, string timestamp, string nonce)
    {
        var method = request.Method.ToUpper();
        var path = request.Path;
        var query = request.QueryString.ToString();
        var body = request.Body;
        
        var data = $"{method}|{path}|{query}|{body}|{timestamp}|{nonce}";
        using (var hmac = new HMACSHA256(Encoding.UTF8.GetBytes(_secretKey)))
        {
            var hash = hmac.ComputeHash(Encoding.UTF8.GetBytes(data));
            return Convert.ToBase64String(hash);
        }
    }
}
```

### 3.2 限流防护
```csharp
public class RateLimitMiddleware
{
    public async Task InvokeAsync(HttpContext context)
    {
        var userId = context.User.FindFirst(ClaimTypes.NameIdentifier)?.Value;
        var ip = context.Connection.RemoteIpAddress?.ToString();
        var endpoint = context.Request.Path;
        
        // 全局限流
        var globalKey = $"rate:global:{ip}";
        var globalCount = await _cache.IncrementAsync(globalKey);
        if (globalCount == 1) await _cache.ExpireAsync(globalKey, TimeSpan.FromSeconds(1));
        if (globalCount > 1000)  // 每秒1000请求
        {
            context.Response.StatusCode = 429;
            await context.Response.WriteAsync("Too many requests");
            return;
        }
        
        // 用户级限流
        if (!string.IsNullOrEmpty(userId))
        {
            var userKey = $"rate:user:{userId}";
            var userCount = await _cache.IncrementAsync(userKey);
            if (userCount == 1) await _cache.ExpireAsync(userKey, TimeSpan.FromMinutes(1));
            if (userCount > 100)  // 每分钟100请求
            {
                context.Response.StatusCode = 429;
                return;
            }
        }
        
        // 接口级限流
        var endpointKey = $"rate:endpoint:{ip}:{endpoint}";
        var endpointCount = await _cache.IncrementAsync(endpointKey);
        if (endpointCount == 1) await _cache.ExpireAsync(endpointKey, TimeSpan.FromMinutes(1));
        if (endpointCount > 60)  // 每分钟60请求
        {
            context.Response.StatusCode = 429;
            return;
        }
        
        await _next(context);
    }
}
```

```csharp
public class EncryptionService
{
    // AES加密
    public string Encrypt(string plainText, string key)
    {
        using (var aes = Aes.Create())
        {
            aes.Key = Encoding.UTF8.GetBytes(key.PadRight(32).Substring(0, 32));
            aes.GenerateIV();
            
            using (var encryptor = aes.CreateEncryptor())
            using (var ms = new MemoryStream())
            {
                ms.Write(aes.IV, 0, aes.IV.Length);
                using (var cs = new CryptoStream(ms, encryptor, CryptoStreamMode.Write))
                using (var sw = new StreamWriter(cs))
                {
                    sw.Write(plainText);
                }
                return Convert.ToBase64String(ms.ToArray());
            }
        }
    }
    
    // 数据库字段加密属性
    public class EncryptedAttribute : Attribute { }
    
    // EF Core Value Converter
    public class EncryptedConverter : ValueConverter<string, string>
    {
        public EncryptedConverter() 
            : base(
                v => EncryptionService.Encrypt(v, Environment.GetEnvironmentVariable("DB_ENCRYPTION_KEY")),
                v => EncryptionService.Decrypt(v, Environment.GetEnvironmentVariable("DB_ENCRYPTION_KEY")))
        { }
    }
}
```

### 4.2 数据脱敏
```csharp
public class DataMaskingService
{
    // 手机号脱敏
    public static string MaskPhone(string phone)
    {
        if (string.IsNullOrEmpty(phone) || phone.Length != 11)
            return phone;
        return phone.Substring(0, 3) + "****" + phone.Substring(7);
    }
    
    // 邮箱脱敏
    public static string MaskEmail(string email)
    {
        if (string.IsNullOrEmpty(email) || !email.Contains("@"))
            return email;
        var parts = email.Split('@');
        var local = parts[0];
        var domain = parts[1];
        var maskedLocal = local.Length > 2 
            ? local.Substring(0, 2) + "***" 
            : local.Substring(0, 1) + "***";
        return $"{maskedLocal}@{domain}";
    }
    
    // 身份证号脱敏
    public static string MaskIdCard(string idCard)
    {
        if (string.IsNullOrEmpty(idCard) || idCard.Length != 18)
            return idCard;
        return idCard.Substring(0, 6) + "********" + idCard.Substring(14);
    }
    
    // IP地址脱敏
    public static string MaskIp(string ip)
    {
        if (string.IsNullOrEmpty(ip)) return ip;
        var parts = ip.Split('.');
        if (parts.Length != 4) return ip;
        return $"***.***.{parts[2]}.{parts[3]}";
    }
}
```

## 5. 内容安全

### 5.1 敏感词过滤
```csharp
public class ContentFilterService
{
    private readonly AhoCorasickTrie _trie;
    
    public ContentFilterService()
    {
        _trie = new AhoCorasickTrie();
        LoadSensitiveWords();
    }
    
    public FilterResult Filter(string content)
    {
        // 1. 检查敏感词
        var matches = _trie.Search(content);
        if (matches.Any())
        {
            return FilterResult.Blocked("Content contains sensitive words");
        }
        
        // 2. 正则匹配
        if (Regex.IsMatch(content, @"\b(?:\d{4}[- ]?\d{4}[- ]?\d{4}[- ]?\d{4})\b"))
        {
            return FilterResult.Blocked("Content contains credit card number");
        }
        
        // 3. AI内容审核
        var aiResult = await _aiContentModeration.CheckAsync(content);
        if (aiResult.IsInappropriate)
        {
            return FilterResult.Blocked($"Content flagged: {aiResult.Reason}");
        }
        
        return FilterResult.Passed();
    }
}
```

### 5.2 文件上传安全
```csharp
public class FileUploadSecurity
{
    // 允许的文件类型
    private static readonly HashSet<string> AllowedVideoTypes = new()
    {
        "video/mp4",
        "video/avi",
        "video/mov",
        "video/mkv"
    };
    
    // 文件扩展名白名单
    private static readonly HashSet<string> AllowedExtensions = new()
    {
        ".mp4", ".avi", ".mov", ".mkv", ".webm"
    };
    
    public async Task<UploadResult> ValidateAndUploadAsync(IFormFile file)
    {
        // 1. 文件大小检查
        if (file.Length > 2L * 1024 * 1024 * 1024)  // 2GB
        {
            return UploadResult.Failed("File too large");
        }
        
        // 2. MIME类型检查
        if (!AllowedVideoTypes.Contains(file.ContentType.ToLower()))
        {
            return UploadResult.Failed("Invalid file type");
        }
        
        // 3. 扩展名检查
        var extension = Path.GetExtension(file.FileName).ToLower();
        if (!AllowedExtensions.Contains(extension))
        {
            return UploadResult.Failed("Invalid file extension");
        }
        
        // 4. 文件头检查（防止伪装）
        using (var stream = file.OpenReadStream())
        {
            var header = new byte[8];
            await stream.ReadAsync(header, 0, 8);
            
            if (!IsValidVideoHeader(header))
            {
                return UploadResult.Failed("Invalid file content");
            }
        }
        
        // 5. 病毒扫描
        var scanResult = await _virusScanner.ScanAsync(file);
        if (!scanResult.IsClean)
        {
            return UploadResult.Failed("File contains malware");
        }
        
        // 6. 上传到隔离区
        var fileName = GenerateSecureFileName(file.FileName);
        await _storage.UploadToQuarantineAsync(file, fileName);
        
        return UploadResult.Success(fileName);
    }
    
    private bool IsValidVideoHeader(byte[] header)
    {
        // MP4: 00 00 00 18 66 74 79 70 (ftyp)
        if (header[4] == 0x66 && header[5] == 0x74 && header[6] == 0x79 && header[7] == 0x70)
            return true;
        
        // AVI: RIFF....AVI
        if (header[0] == 0x52 && header[1] == 0x49 && header[2] == 0x46 && header[3] == 0x46)
            return true;
        
        return false;
    }
}
```

### 6.2 密钥管理
```csharp
public class SecretManager
{
    // 使用Azure Key Vault / AWS Secrets Manager
    public async Task<string> GetSecretAsync(string key)
    {
        // 优先从环境变量获取
        var envValue = Environment.GetEnvironmentVariable(key);
        if (!string.IsNullOrEmpty(envValue))
            return envValue;
        
        // 从Key Vault获取
        return await _keyVault.GetSecretAsync(key);
    }
    
    // 密钥轮换
    public async Task RotateKeysAsync()
    {
        // JWT密钥轮换
        var newKey = GenerateSecureKey();
        await _keyVault.SetSecretAsync("Jwt-Key-New", newKey);
        
        // 等待旧Token过期（2小时）
        await Task.Delay(TimeSpan.FromHours(2.5));
        
        // 替换旧密钥
        await _keyVault.SetSecretAsync("Jwt-Key", newKey);
        await _keyVault.DeleteSecretAsync("Jwt-Key-New");
    }
}
```
