## 3. 核心表结构

### 3.1 用户相关

#### Users表
```sql
CREATE TABLE "Users" (
    "Id" uuid PRIMARY KEY,
    "UserName" varchar(256) NOT NULL,
    "NormalizedUserName" varchar(256),
    "Email" varchar(256),
    "NormalizedEmail" varchar(256),
    "EmailConfirmed" boolean DEFAULT false,
    "PasswordHash" varchar(256),
    "PhoneNumber" varchar(20),
    "PhoneNumberConfirmed" boolean DEFAULT false,
    "TwoFactorEnabled" boolean DEFAULT false,
    "LockoutEnd" timestamp with time zone,
    "LockoutEnabled" boolean DEFAULT true,
    "AccessFailedCount" integer DEFAULT 0,
    "Avatar" varchar(500),
    "Signature" varchar(500),
    "Level" integer DEFAULT 0,
    "Experience" integer DEFAULT 0,
    "Gender" integer,
    "BirthDate" date,
    "IsVerified" boolean DEFAULT false,
    "FollowerCount" integer DEFAULT 0,
    "FollowingCount" integer DEFAULT 0,
    "IsDeleted" boolean DEFAULT false,
    "CreationTime" timestamp with time zone NOT NULL,
    "LastModificationTime" timestamp with time zone
);

CREATE UNIQUE INDEX "IX_Users_UserName" ON "Users"("UserName") WHERE "IsDeleted" = false;
CREATE UNIQUE INDEX "IX_Users_Email" ON "Users"("Email") WHERE "IsDeleted" = false AND "Email" IS NOT NULL;
CREATE INDEX "IX_Users_CreationTime" ON "Users"("CreationTime" DESC) WHERE "IsDeleted" = false;
```

#### UserProfiles表
```sql
CREATE TABLE "UserProfiles" (
    "Id" uuid PRIMARY KEY,
    "UserId" uuid NOT NULL UNIQUE REFERENCES "Users"("Id"),
    "InterestTags" jsonb DEFAULT '[]',
    "PreferredCategories" uuid[],
    "WatchPatterns" jsonb,
    "ActiveLevel" integer DEFAULT 0,
    "LastUpdated" timestamp with time zone NOT NULL,
    "CreationTime" timestamp with time zone NOT NULL
);

CREATE INDEX "IX_UserProfiles_UserId" ON "UserProfiles"("UserId");
```

### 3.2 视频相关

#### Videos表
```sql
CREATE TABLE "Videos" (
    "Id" uuid PRIMARY KEY,

    "Title" varchar(200) NOT NULL,

    "CoverUrl" varchar(500),
    "CategoryId" uuid NOT NULL REFERENCES "Categories"("Id"),
    "Tags" jsonb DEFAULT '[]',

    "Width" integer,
    "Height" integer,
    "Resolutions" jsonb,
    "MasterPlaylistUrl" varchar(500),
    "OriginalFileUrl" varchar(500),
    "FileHash" varchar(64),
    "ViewCount" bigint DEFAULT 0,
    "LikeCount" bigint DEFAULT 0,
    "CoinCount" bigint DEFAULT 0,
    "CollectCount" bigint DEFAULT 0,
    "DanmakuCount" bigint DEFAULT 0,
    "CommentCount" bigint DEFAULT 0,
    "PublishTime" timestamp with time zone,
    "IsDeleted" boolean DEFAULT false,
    "CreationTime" timestamp with time zone NOT NULL,
    "LastModificationTime" timestamp with time zone

CREATE INDEX "IX_Videos_UserId" ON "Videos"("UserId") WHERE "IsDeleted" = false;
```

### 3.3 弹幕相关

#### Danmakus表
```sql
CREATE TABLE "Danmakus" (
    "Id" uuid PRIMARY KEY,
    "VideoId" uuid NOT NULL REFERENCES "Videos"("Id"),
    "UserId" uuid REFERENCES "Users"("Id"),
    "Content" varchar(200) NOT NULL,
    "Time" decimal(10, 3) NOT NULL,
    "Type" integer DEFAULT 0,  -- 0=滚动,1=顶部,2=底部
    "Color" varchar(7) DEFAULT '#FFFFFF',
    "Size" integer DEFAULT 24,
    "Mode" integer DEFAULT 0,
    "IsAdvanced" boolean DEFAULT false,
    "IP" varchar(50) NOT NULL,
    "IsDeleted" boolean DEFAULT false,
    "CreationTime" timestamp with time zone NOT NULL
) PARTITION BY RANGE ("CreationTime");

-- 按月份分区（弹幕量巨大，必须分区）
CREATE TABLE "Danmakus_2024_01" PARTITION OF "Danmakus"
FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

CREATE INDEX "IX_Danmakus_VideoId_Time" ON "Danmakus"("VideoId", "Time") 
WHERE "IsDeleted" = false;
CREATE INDEX "IX_Danmakus_UserId" ON "Danmakus"("UserId");
CREATE INDEX "IX_Danmakus_CreationTime" ON "Danmakus"("CreationTime" DESC);
```

## 4. 数据库优化策略

### 4.1 索引策略
```sql
-- 复合索引示例
CREATE INDEX "IX_Videos_Status_PublishTime" 
ON "Videos"("Status", "PublishTime" DESC) 
WHERE "IsDeleted" = false;

-- 部分索引（PostgreSQL特性）
CREATE INDEX "IX_Videos_Published" 
ON "Videos"("PublishTime" DESC) 
WHERE "Status" = 2 AND "IsDeleted" = false;

-- GIN索引（用于JSON数组搜索）
CREATE INDEX "IX_Videos_Tags_Gin" 
ON "Videos" USING GIN("Tags");
```

### 4.2 分区策略
```sql
-- 按时间范围分区（适用于弹幕、评论等海量数据）
CREATE TABLE "Danmakus" (
    "Id" uuid NOT NULL,
    "VideoId" uuid NOT NULL,
    "UserId" uuid,
    "Content" varchar(200) NOT NULL,
    "Time" decimal(10, 3) NOT NULL,
    "CreationTime" timestamp with time zone NOT NULL,
    PRIMARY KEY ("Id", "CreationTime")
) PARTITION BY RANGE ("CreationTime");

-- 创建分区
CREATE TABLE "Danmakus_2024_01" PARTITION OF "Danmakus"
FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

CREATE TABLE "Danmakus_2024_02" PARTITION OF "Danmakus"
FOR VALUES FROM ('2024-02-01') TO ('2024-03-01');
```

## 5. Redis数据设计

### 5.1 缓存Key规范
| Key | 类型 | 说明 | TTL |
|-----|------|------|-----|
| `user:{userId}` | String | 用户信息 | 10分钟 |
| `user:follow:{userId}` | Set | 关注列表 | 5分钟 |
| `user:fans:{userId}` | Set | 粉丝列表 | 5分钟 |
| `video:{videoId}` | String | 视频信息 | 10分钟 |
| `video:hot` | SortedSet | 热门视频 | 1分钟 |
| `video:count:{videoId}` | Hash | 计数器 | 永久 |
| `danmaku:{videoId}:{start}:{end}` | String | 弹幕缓存 | 5分钟 |
| `search:suggest:{keyword}` | List | 搜索建议 | 1小时 |

### 5.2 计数器设计

```
video:count:{videoId}
```
