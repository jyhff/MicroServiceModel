// VideoServiceTests.cs
public class VideoServiceTests
{
    private readonly Mock<IVideoRepository> _videoRepoMock;
    private readonly Mock<IUserRepository> _userRepoMock;
    private readonly Mock<ITranscodeService> _transcodeMock;
    private readonly VideoService _service;

    public VideoServiceTests()
    {
        _videoRepoMock = new Mock<IVideoRepository>();
        _userRepoMock = new Mock<IUserRepository>();
        _transcodeMock = new Mock<ITranscodeService>();
        
        _service = new VideoService(
            _videoRepoMock.Object,
            _userRepoMock.Object,
            _transcodeMock.Object);
    }

    [Fact]
    public async Task CreateVideo_WithValidInput_ReturnsVideo()
    {
        // Arrange
        var input = new CreateVideoInput
        {
            Title = "Test Video",
            Description = "Description",
            CategoryId = Guid.NewGuid(),
            UserId = Guid.NewGuid()
        };

        _videoRepoMock.Setup(x => x.InsertAsync(It.IsAny<Video>()))
            .Returns(Task.CompletedTask);

        // Act
        var result = await _service.CreateAsync(input);

        // Assert
        result.Should().NotBeNull();
        result.Title.Should().Be(input.Title);
        _videoRepoMock.Verify(x => x.InsertAsync(It.IsAny<Video>()), Times.Once);
    }

    [Fact]
    public async Task CreateVideo_WithEmptyTitle_ThrowsValidationException()
    {
        // Arrange
        var input = new CreateVideoInput
        {
            Title = "",
            UserId = Guid.NewGuid()
        };

        // Act & Assert
        await Assert.ThrowsAsync<ValidationException>(() => 
            _service.CreateAsync(input));
    }

    [Fact]
    public async Task GetVideo_WithExistingId_ReturnsVideo()
    {
        // Arrange
        var videoId = Guid.NewGuid();
        var video = new Video { Id = videoId, Title = "Test" };
        
        _videoRepoMock.Setup(x => x.GetAsync(videoId))
            .ReturnsAsync(video);

        // Act
        var result = await _service.GetAsync(videoId);

        // Assert
        result.Should().NotBeNull();
        result.Id.Should().Be(videoId);
    }

    [Fact]
    public async Task GetVideo_WithNonExistingId_ReturnsNull()
    {
        // Arrange
        var videoId = Guid.NewGuid();
        _videoRepoMock.Setup(x => x.GetAsync(videoId))
            .ReturnsAsync((Video)null);

        // Act
        var result = await _service.GetAsync(videoId);

        // Assert
        result.Should().BeNull();
    }
}

public class IntegrationTestsFixture : IAsyncLifetime
{
    public WebApplicationFactory<Program> Factory { get; private set; }
    public HttpClient Client { get; private set; }

    public async Task InitializeAsync()
    {
        Factory = new WebApplicationFactory<Program>()
            .WithWebHostBuilder(builder =>
            {
                builder.ConfigureServices(services =>
                {
                    // 替换为测试数据库
                    services.RemoveAll<DbContextOptions<ApplicationDbContext>>();
                    services.AddDbContext<ApplicationDbContext>(options =>
                        options.UseNpgsql("Host=localhost;Database=TestDB"));
                    
                    // 使用内存缓存
                    services.AddDistributedMemoryCache();
                });
            });

        Client = Factory.CreateClient();
        
        // 初始化数据库
        using var scope = Factory.Services.CreateScope();
        var db = scope.ServiceProvider.GetRequiredService<ApplicationDbContext>();
        await db.Database.MigrateAsync();
    }

    public async Task DisposeAsync()
    {
        Client?.Dispose();
        Factory?.Dispose();
    }
}

// VideoApiTests.cs
public class VideoApiTests : IClassFixture<IntegrationTestsFixture>
{
    private readonly HttpClient _client;
    private readonly IntegrationTestsFixture _fixture;

    public VideoApiTests(IntegrationTestsFixture fixture)
    {
        _fixture = fixture;
        _client = fixture.Client;
    }

    [Fact]
    public async Task GetVideos_ReturnsOk()
    {
        // Act
        var response = await _client.GetAsync("/api/videos");

        // Assert
        response.StatusCode.Should().Be(HttpStatusCode.OK);
        
        var content = await response.Content.ReadAsStringAsync();
        var result = JsonSerializer.Deserialize<PagedResult<VideoDto>>(content);
        result.Should().NotBeNull();
        result.Items.Should().NotBeNull();
    }

    [Fact]
    public async Task CreateVideo_WithAuth_ReturnsCreated()
    {
        // Arrange
        var token = await GetAuthTokenAsync();
        _client.DefaultRequestHeaders.Authorization = 
            new AuthenticationHeaderValue("Bearer", token);

        var input = new { Title = "Test Video", CategoryId = Guid.NewGuid() };
        var json = JsonSerializer.Serialize(input);

        // Act
        var response = await _client.PostAsync(
            "/api/videos",
            new StringContent(json, Encoding.UTF8, "application/json"));

        // Assert
        response.StatusCode.Should().Be(HttpStatusCode.Created);
    }

    private async Task<string> GetAuthTokenAsync()
    {
        // 获取测试token
        var response = await _client.PostAsJsonAsync("/api/auth/login", new
        {
            Account = "test@test.com",
            Password = "test123"
        });
        
        var result = await response.Content.ReadFromJsonAsync<LoginResult>();
        return result.AccessToken;
    }
}

import { test, expect } from '@playwright/test';

test.describe('视频上传流程', () => {

// load-test.js
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  stages: [
    { duration: '2m', target: 100 },    // 预热
    { duration: '5m', target: 100 },  // 稳定
    { duration: '2m', target: 200 },    // 峰值
    { duration: '5m', target: 200 },    // 峰值稳定
    { duration: '2m', target: 0 },      // 冷却
  ],
  thresholds: {
    http_req_duration: ['p(95)<500'],    // 95%请求<500ms
    http_req_failed: ['rate<0.01'],      // 错误率<1%
  },
};

export default function () {
  const res = http.get('https://api.bilibili.com/api/videos');
  
  check(res, {
    'status is 200': (r) => r.status === 200,
    'response time < 500ms': (r) => r.timings.duration < 500,
  });
  
  sleep(1);
}

  });

### 6.2 CI/CD集成
```yaml
# .github/workflows/test.yml
name: Test

on: [push, pull_request]

jobs:
  unit-tests:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup .NET
        uses: actions/setup-dotnet@v3
        with:
          dotnet-version: '10.0.x'
      
      - name: Restore dependencies
        run: dotnet restore
      
      - name: Build
        run: dotnet build --no-restore
      
      - name: Run tests
        run: dotnet test --no-build --verbosity normal
      
      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          files: ./coverage.xml

  e2e-tests:
    runs-on: ubuntu-latest
    needs: unit-tests
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node
        uses: actions/setup-node@v3
      
      - name: Install Playwright
        run: npm ci && npx playwright install
      
      - name: Run E2E tests
        run: npx playwright test
      
      - name: Upload results
        uses: actions/upload-artifact@v3
        with:
          name: playwright-report
          path: playwright-report/
```
